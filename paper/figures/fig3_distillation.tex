\documentclass[tikz,border=4pt]{standalone}
\usepackage{tikz}
\usetikzlibrary{arrows.meta,positioning,shapes.geometric,fit,backgrounds,decorations.pathreplacing}
\usepackage{xcolor}

\definecolor{teacher}{RGB}{244,67,54}
\definecolor{student}{RGB}{33,150,243}
\definecolor{ensemble}{RGB}{156,39,176}
\definecolor{softlabel}{RGB}{255,152,0}

\begin{document}
\begin{tikzpicture}[
  model/.style={draw, rounded corners=4pt, minimum width=2.2cm, minimum height=0.8cm,
                font=\small\bfseries, align=center, fill=#1!15, draw=#1!70},
  prob/.style={draw, rounded corners=2pt, minimum width=1.4cm, minimum height=0.55cm,
               font=\scriptsize, align=center, fill=#1!12, draw=#1!60},
  arrow/.style={-Stealth, thick},
  >=Stealth, node distance=0.7cm
]

%% ── Teacher models (top row) ─────────────────────────────────────────────────
\node[model=teacher] (t1) at (0, 2.5) {Teacher 1\\(v5b seed 42)};
\node[model=teacher, right=0.5cm of t1] (t2) {Teacher 2\\(v5b seed 123)};
\node[model=teacher, right=0.5cm of t2] (t3) {Teacher 3\\(v5b seed 7)};
\node[model=teacher, right=0.5cm of t3] (t4) {Teacher 4\\(v9, frozen BN)};

%% ── Probability outputs ──────────────────────────────────────────────────────
\node[prob=teacher, below=0.5cm of t1] (p1) {$\mathbf{p}^{(1)}$};
\node[prob=teacher, below=0.5cm of t2] (p2) {$\mathbf{p}^{(2)}$};
\node[prob=teacher, below=0.5cm of t3] (p3) {$\mathbf{p}^{(3)}$};
\node[prob=teacher, below=0.5cm of t4] (p4) {$\mathbf{p}^{(4)}$};

%% ── Ensemble average ─────────────────────────────────────────────────────────
\node[prob=ensemble, below=0.7cm of p2, xshift=1.2cm,
      minimum width=2.4cm, font=\small] (ens)
  {$\mathbf{p}^{\text{teach}} = \frac{1}{4}\sum_k \mathbf{p}^{(k)}$};

%% ── Student ──────────────────────────────────────────────────────────────────
\node[model=student, below=2.2cm of ens] (student)
  {Student\\\textsc{TransFuserV5-KD}};

%% ── Student output ───────────────────────────────────────────────────────────
\node[prob=student, right=0.8cm of student] (ps)
  {$\mathbf{p}^s$};

%% ── Loss nodes ───────────────────────────────────────────────────────────────
\node[draw, fill=softlabel!20, draw=softlabel!70, rounded corners=3pt,
      minimum width=2.0cm, minimum height=0.6cm, font=\scriptsize\bfseries,
      right=1.0cm of ens, yshift=-1.0cm] (kd_loss)
  {$\mathcal{L}_\text{KD}$\\KL div ($\tau=4$)};

\node[draw, fill=gray!15, draw=gray!60, rounded corners=3pt,
      minimum width=2.0cm, minimum height=0.6cm, font=\scriptsize\bfseries,
      below=0.3cm of kd_loss] (hard_loss)
  {$\mathcal{L}_\text{focal}$\\hard labels};

\node[draw, fill=green!15, draw=green!60, rounded corners=3pt,
      minimum width=2.0cm, minimum height=0.6cm, font=\scriptsize\bfseries,
      below=0.3cm of hard_loss] (ord_loss)
  {$\mathcal{L}_\text{ord}$\\ordinal distance};

%% ── Total loss ────────────────────────────────────────────────────────────────
\node[draw, fill=red!12, draw=red!60, rounded corners=3pt,
      minimum width=2.2cm, minimum height=0.65cm, font=\small\bfseries,
      right=0.9cm of hard_loss] (total)
  {$\mathcal{L}_\text{total}$};

%% ── Input data ────────────────────────────────────────────────────────────────
\node[draw, fill=gray!10, rounded corners=3pt, minimum width=2.0cm,
      minimum height=0.55cm, font=\scriptsize, left=1.0cm of student] (data)
  {Multimodal input\\$\mathbf{X}_{t-4:t}$};

%% ── Arrows ────────────────────────────────────────────────────────────────────
\foreach \t/\p in {t1/p1, t2/p2, t3/p3, t4/p4}{
  \draw[arrow, teacher] (\t) -- (\p);
}
\draw[arrow, ensemble] (p1.south) -- (ens.north west);
\draw[arrow, ensemble] (p2.south) -- (ens.north);
\draw[arrow, ensemble] (p3.south) -- (ens.north east);
\draw[arrow, ensemble] (p4.south) -- (ens.east |- ens.north);
\draw[arrow] (data) -- (student);
\draw[arrow, student] (student) -- (ps);
\draw[arrow, softlabel] (ens.south) -- (kd_loss.north);
\draw[arrow, softlabel] (ps.east) -| (kd_loss.west);
\draw[arrow] (ps.east) -| (hard_loss.west);
\draw[arrow] (ps.east) -| (ord_loss.west);
\draw[arrow] (kd_loss.east) -- (total.west |- kd_loss);
\draw[arrow] (hard_loss.east) -- (total.west);
\draw[arrow] (ord_loss.east) -- (total.west |- ord_loss);

%% ── Weight labels ─────────────────────────────────────────────────────────────
\node[font=\scriptsize, right=0.05cm of kd_loss.east,
      xshift=0.0cm, text=softlabel!80] at ($(kd_loss.east)!0.5!(total.west |- kd_loss)$)
  {$\alpha=0.7$};
\node[font=\scriptsize] at ($(hard_loss.east)!0.5!(total.west)$)
  {$1{-}\alpha$};
\node[font=\scriptsize] at ($(ord_loss.east)!0.5!(total.west |- ord_loss)$)
  {$\lambda=0.5$};

%% ── Backprop arrow ────────────────────────────────────────────────────────────
\draw[arrow, dashed, thick, gray] (total.south) |- (student.east)
  node[midway, font=\scriptsize, text=gray, below right=0.1cm and 0.1cm] {backprop (student only)};

%% ── Frozen badge ─────────────────────────────────────────────────────────────
\node[draw, fill=gray!20, rounded corners=2pt, font=\scriptsize,
      above left=0.3cm and 0.1cm of t1] (frz_badge) {\textit{frozen at inference}};
\draw[gray, ->] (frz_badge.east) -- (t1.north west);

\end{tikzpicture}
\end{document}
